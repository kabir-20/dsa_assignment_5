#include <iostream>
using namespace std;


class Node {
public:
    int coeff;   
    int pow;     
    Node* next;

    Node(int c, int p) {
        coeff = c;
        pow = p;
        next = NULL;
    }
};


class Polynomial {
public:
    Node* head;

    Polynomial() {
        head = NULL;
    }


    void insertTerm(int coeff, int pow) {
        Node* newNode = new Node(coeff, pow);

        if (head == NULL) {
            head = newNode;
            return;
        }

        Node* temp = head;
        while (temp->next != NULL)
            temp = temp->next;

        temp->next = newNode;
    }


    static Polynomial Polynomials(Polynomial p1, Polynomial p2) {
        Polynomial result;
        Node* ptr1 = p1.head;
        Node* ptr2 = p2.head;

        while (ptr1 != NULL && ptr2 != NULL) {
          
            if (ptr1->pow == ptr2->pow) {
                result.insertTerm(ptr1->coeff + ptr2->coeff, ptr1->pow);
                ptr1 = ptr1->next;
                ptr2 = ptr2->next;
            }
         
            else if (ptr1->pow > ptr2->pow) {
                result.insertTerm(ptr1->coeff, ptr1->pow);
                ptr1 = ptr1->next;
            }
       
            else {
                result.insertTerm(ptr2->coeff, ptr2->pow);
                ptr2 = ptr2->next;
            }
        }

      
        while (ptr1 != NULL) {
            result.insertTerm(ptr1->coeff, ptr1->pow);
            ptr1 = ptr1->next;
        }

       
        while (ptr2 != NULL) {
            result.insertTerm(ptr2->coeff, ptr2->pow);
            ptr2 = ptr2->next;
        }

        return result;
    }

   
    void display() {
        Node* temp = head;
        while (temp != NULL) {
            cout << temp->coeff << "x^" << temp->pow;
            if (temp->next != NULL)
                cout << " + ";
            temp = temp->next;
        }
        cout << endl;
    }
};


int main() {
    Polynomial p1, p2, result;

  
    p1.insertTerm(5, 2);
    p1.insertTerm(4, 1);
    p1.insertTerm(2, 0);


    p2.insertTerm(5, 1);
    p2.insertTerm(5, 0);

    cout << "First Polynomial: ";
    p1.display();

    cout << "Second Polynomial: ";
    p2.display();

    result = Polynomial::addPolynomials(p1, p2);

    cout << "Resultant Polynomial: ";
    result.display();

    return 0;
}
